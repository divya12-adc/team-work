  </form>

  <div id="attendance-list">
    <h3>Attendance Records</h3>
    <div id="records"></div>
  </div>

  <script>
    const form = document.getElementById('attendance-form');
    const recordsDiv = document.getElementById('records');

    // Load existing records
    window.onload = () => {
      const stored = JSON.parse(localStorage.getItem('attendance')) || [];
      stored.forEach(r => addRecordToDOM(r.name, r.time));
    };

    form.addEventListener('submit', function(e) {
      e.preventDefault();
      const nameInput = document.getElementById('student-name');
      const name = nameInput.value.trim();

      if (name !== "") {
        const time = new Date().toLocaleString();
        addRecordToDOM(name, time);

        // Save to local storage
        const records = JSON.parse(localStorage.getItem('attendance')) || [];
        records.push({ name, time });
        localStorage.setItem('attendance', JSON.stringify(records));

        nameInput.value = '';
      }
    });

    function addRecordToDOM(name, time) {
      const div = document.createElement('div');
      div.className = 'record';
      div.innerHTML = `<strong>${name}</strong> - <em>${time}</em>`;
      recordsDiv.appendChild(div);
    }
  </script>
